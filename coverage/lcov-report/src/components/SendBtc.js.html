import React, { Component } from 'react';
 
import Resumen from './Resumen';
import MontoTotal from './MontoTotal';
import Loading from './Loading';
 
export default class SendBtc extends Component {
 
    state = {
        totalAmount: 2000,
        address: '',
        amount: 0,
        fee: parseFloat((Math.random() / 1000).toFixed(5)),
        errorAddress: false,
        errorAmount: false,
        envios: []
    }
 
    checkForm = (addressBtc, amount) => { //checkea errores del formulario
        // checkerrores
        if (addressBtc !== 'address correcto') {
            setTimeout(() => {
                console.error('address incorrecto')
                this.setState({
                    errorAddress: true
                })
            }, 500)
            return false
        } else if (amount > this.state.totalAmount) {
            console.error('no puede exceder su monto')
            this.setState({
                errorAmount: true
            })
            return false
        } else if (amount === 0){
            console.error('ingrese un monto')
            this.setState({
                errorAmount: true
            })
            return false
        } else {
            return true
        }
    }
 
    sendBitcoin = data => { //envia bitcoin
        this.setState({
            loading: true
        })
 
        setTimeout(() => {
          this.setState( prevState => ({
            address: '',
            amount: 0,
            errorAddress: false,
            errorAmount: false,
            envios: [...prevState.envios, data],
            totalAmount: (prevState.totalAmount - data.amount).toFixed(5),
            loading: false
          }))
        }, 1500)
    }
    
    handleOnSubmit = e => { // trae los datos del form, los checkea y envia
        e.preventDefault();
        let addressBtc = this.state.address;
        let amount = parseFloat(this.state.amount) + this.state.fee;
 
        if (this.checkForm(addressBtc, amount)) {
            let data = {
                addressBtc,
                amount: amount + this.state.fee,
                totalAmount: this.state.totalAmount - (amount + this.state.fee)
            }
            this.sendBitcoin(data)
        } else {
            console.error('error al realizar el envio')
        }
    }
 
    handleOnChange = e => { //maneja los values con el state
        this.setState({
            [e.target.name]: e.target.value
        })
    }
 
    render() {    
        return (
            <div className="container">
                <MontoTotal totalAmount={this.state.totalAmount}/>
 
                { !this.state.loading ? 
                    <form onSubmit={this.handleOnSubmit} id="form">
                        <div className="form-group">
                            <label>Address BTC</label>
                            <input
                                className="input"
                                placeholder="address btc"
                                name="address"
                                value={this.state.address} 
                                onChange={this.handleOnChange}/>
                        </div>
                        
                        <div className="form-group">
                            <label>Amount to send</label>
                            <input
                                className="input"
                                placeholder="amount to send"
                                name="amount"
                                value={this.state.amount}
                                onChange={this.handleOnChange}/>
                            <small>Fee: {this.state.fee}</small>
                        </div>
                        
                        <button
                            className="button btn-azul submit"
                            type="submit">Enviar</button>
                    </form>
                :
                    <Loading/>
                }
 
                
 
                { this.state.envios.map( (envio, index) => {
                    return (
                        <Resumen
                            key={index}
                            address={envio.addressBtc}
                            amount={envio.amount}
                            totalAmount={envio.totalAmount}/>
                        )
                    })
                }
                
            </div>
        )
    }
}
